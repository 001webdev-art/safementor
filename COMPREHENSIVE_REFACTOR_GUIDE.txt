      /**
       * Environment variable validation and access layer.
       * Ensures all required variables are present at startup.
       * Fails fast with helpful error messages.
       */

      interface EnvConfig {
        NEXT_PUBLIC_SUPABASE_URL: string;
        NEXT_PUBLIC_SUPABASE_ANON_KEY: string;
        SITE_PASSWORD: string;
      }

      function getEnvVar(key: keyof EnvConfig): string {
        const value = process.env[key];
        if (!value) {
          throw new Error(
            `Missing environment variable: ${key}\n` +
            `Please add ${key} to your .env.local file.\n\n` +
            `Required variables:\n` +
            `  - NEXT_PUBLIC_SUPABASE_URL\n` +
            `  - NEXT_PUBLIC_SUPABASE_ANON_KEY\n` +
            `  - SITE_PASSWORD`
          );
        }
        return value;
      }

      /**
       * Validated environment variables.
       * Access via: import { env } from '@/lib'
       * 
       * This object is created at module load time, so any missing
       * variables will cause an immediate error.
       */
      export const env: EnvConfig = {
        NEXT_PUBLIC_SUPABASE_URL: getEnvVar('NEXT_PUBLIC_SUPABASE_URL'),
        NEXT_PUBLIC_SUPABASE_ANON_KEY: getEnvVar('NEXT_PUBLIC_SUPABASE_ANON_KEY'),

    export interface Message {
      role: 'user' | 'assistant';
      content: string;
    }

    export interface ProviderConfig {
      apiKey: string;
      model?: string;
    }

    export interface ProviderResponse {
      text: string;
      provider: string;
      tokensUsed?: {
        input: number;
        output: number;
      };
    }

    export abstract class LLMProvider {
      protected config: ProviderConfig;
      protected abstract model: string;

      constructor(config: ProviderConfig) {
        this.config = config;
      }

      abstract sendMessage(
        messages: Message[],
        systemPrompt?: string
      ): Promise<ProviderResponse>;

      protected parseResponse(text: string): ProviderResponse {
        return {
          text,
          provider: this.constructor.name,
        };
      }
    }

STEP 5: Create Provider Registry

  FILE: src/lib/llm/providers/registry.ts
  
  Content:
    import { GeminiProvider } from './gemini';
    import { GPTProvider } from './gpt';
    import { GroqProvider } from './groq';
    import { LLMProvider, ProviderConfig } from './base';

    type ProviderType = 'gemini' | 'gpt' | 'groq';

    export class ProviderRegistry {
      private static providers: Record<
        ProviderType,
        new (config: ProviderConfig) => LLMProvider
      > = {
        gemini: GeminiProvider,
        gpt: GPTProvider,
        groq: GroqProvider,
      };

      static getProvider(
        type: ProviderType,
        config: ProviderConfig
      ): LLMProvider {
        const ProviderClass = this.providers[type];
        if (!ProviderClass) {
          throw new Error(`Unknown provider: ${type}`);
        }
        return new ProviderClass(config);
      }

      static getAvailableProviders(): ProviderType[] {
        return Object.keys(this.providers) as ProviderType[];
      }
    }

    import { ProviderRegistry } from '@/lib/llm/providers/registry';
    import { Message } from '@/lib/llm/providers/base';

    export async function POST(request: Request) {
      try {
        const body = await request.json();
        const { provider, messages, systemPrompt } = body;

        // Validate request
        if (!provider || !messages || !Array.isArray(messages)) {
          return Response.json(
            {
              error: 'Missing required fields: provider, messages',
              example: {
                provider: 'gemini | gpt | groq',
                messages: [{ role: 'user', content: 'Hello' }],
                systemPrompt: 'Optional system instruction',
              },
            },
            { status: 400 }
          );
        }

        // Get API key for provider
        const apiKeyEnvVar = `${provider.toUpperCase()}_API_KEY`;
        const apiKey = process.env[apiKeyEnvVar];

        if (!apiKey) {
          return Response.json(
            { error: `Missing API key: ${apiKeyEnvVar}` },
            { status: 500 }
          );
        }

        // Get provider and send message
        const llmProvider = ProviderRegistry.getProvider(provider as any, {
          apiKey,
        });

        const response = await llmProvider.sendMessage(
          messages as Message[],
          systemPrompt
        );

        return Response.json(response);
      } catch (error) {
        console.error('Chat API error:', error);
        return Response.json(
          {
            error: error instanceof Error ? error.message : 'Failed to process request',
          },
          { status: 500 }
        );
      }
    }

      state = { hasError: false, error: undefined };

      static getDerivedStateFromError(error: Error) {
        return { hasError: true, error };
      }

      componentDidCatch(error: Error, errorInfo: React.ErrorInfo) {
        console.error('Error caught by boundary:', error, errorInfo);
      }

      render() {
        if (this.state.hasError) {
          return (
            this.props.fallback || (
              <div className="p-4 bg-red-50 border border-red-200 rounded-lg">
                <h2 className="font-bold text-red-800">Something went wrong</h2>
                <p className="text-red-700">{this.state.error?.message}</p>
                <button
                  onClick={() => this.setState({ hasError: false })}
                  className="mt-2 px-4 py-2 bg-red-600 text-white rounded"
                >
                  Try again
                </button>
              </div>
            )
          );
        }

        return this.props.children;
      }
    }

  Usage:
    <ErrorBoundary>
      <DashboardClient />
    </ErrorBoundary>

  Benefit:
    ✓ Graceful error handling
    ✓ User sees helpful message
    ✓ Other features still work if one fails
    ✓ Easy debugging

  Time: 1-2 hours

IMPROVEMENT 2: Create API Request Abstraction Layer

  Goal: Centralized API logic and error handling
  
  Create: src/lib/api/client.ts
  
  Content:
    export class ApiClient {
      static async chat(
        messages: Message[],
        provider: string
      ): Promise<ApiResponse<ChatResponse>> {
        try {
          const response = await fetch('/api/chat', {
            method: 'POST',
            headers: { 'Content-Type': 'application/json' },
            body: JSON.stringify({ messages, provider }),
          });

          if (!response.ok) {
            throw new Error(`API error: ${response.status}`);
          }

          const data = await response.json();
          return { data, status: response.status };
        } catch (error) {
          return {
            error: error instanceof Error ? error.message : 'Unknown error',
            status: 500,
          };
        }
      }
    }

  Usage:

    } else {
