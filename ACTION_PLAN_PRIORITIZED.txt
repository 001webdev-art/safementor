TASK 2.3: CREATE ENVIRONMENT VALIDATION MODULE
────────────────────────────────────────────────

Priority: 🟡 MEDIUM
Time: 45 minutes

CREATE: src/lib/config/env.ts

Content Template:
  interface EnvConfig {
    NEXT_PUBLIC_SUPABASE_URL: string;
    NEXT_PUBLIC_SUPABASE_ANON_KEY: string;
    SITE_PASSWORD: string;
  }

  function getEnvVar(key: keyof EnvConfig): string {
    const value = process.env[key];
    if (!value) {
      throw new Error(`Missing environment variable: ${key}`);
    }
    return value;
  }

  export const env: EnvConfig = {
    NEXT_PUBLIC_SUPABASE_URL: getEnvVar('NEXT_PUBLIC_SUPABASE_URL'),
    NEXT_PUBLIC_SUPABASE_ANON_KEY: getEnvVar('NEXT_PUBLIC_SUPABASE_ANON_KEY'),
    SITE_PASSWORD: getEnvVar('SITE_PASSWORD'),
  };

Steps:
  1. CREATE file
  2. UPDATE src/lib/index.ts: export { env } from './config/env';
  3. UPDATE src/middleware.ts: import { env } from '@/lib';
  4. REPLACE: process.env.SITE_PASSWORD → env.SITE_PASSWORD
  5. UPDATE src/app/api/access/route.ts: Same replacement
  6. TEST: npm run dev → Should show error if env var missing
  7. COMMIT
